#include <iostream>
#include <fstream>
#include <string>
using namespace std;

struct Employee {
    int empcode;
    string empname;
    float salary;
};

int main() {
    ofstream outFile("EMP.txt", ios::app);
    int n;
    cout << "How many employees? ";
    cin >> n;

    for(int i = 0; i < n; i++) {
        Employee e;
        cout << "Enter empcode, empname, salary: ";
        cin >> e.empcode >> e.empname >> e.salary;
        outFile.write((char*)&e, sizeof(e));
    }
    outFile.close();

    ifstream inFile("EMP.txt");
    Employee e, highest;
    bool first = true;

    while(inFile.read((char*)&e, sizeof(e))) {
        if(first) {
            highest = e;
            first = false;
        } else if(e.salary > highest.salary) {
            highest = e;
        }
    }
    inFile.close();

    cout << "Employee with highest salary:\n";
    cout << "Code: " << highest.empcode << ", Name: " << highest.empname 
         << ", Salary: " << highest.salary << endl;

    return 0;
}
