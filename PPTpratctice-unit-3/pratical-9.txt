#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main() {
    string arr[5];
    cout << "Enter 5 strings:\n";
    for(int i = 0; i < 5; i++) cin >> arr[i];

    sort(arr, arr + 5);

    cout << "Strings in ascending order:\n";
    for(int i = 0; i < 5; i++) cout << arr[i] << endl;

    return 0;
}
