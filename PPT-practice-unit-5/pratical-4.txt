#include <iostream>
using namespace std;

int main() {
    int N;
    cout << "Enter number of elements: ";
    cin >> N;

    int arr[N];
    cout << "Enter elements: ";
    for(int i = 0; i < N; i++) cin >> arr[i];

    // Bubble sort
    for(int i = 0; i < N-1; i++) {
        for(int j = 0; j < N-i-1; j++) {
            if(arr[j] > arr[j+1]) {
                int temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }

    cout << "Sorted array in ascending order: ";
    for(int i = 0; i < N; i++) cout << arr[i] << " ";
    cout << endl;

    return 0;
}
