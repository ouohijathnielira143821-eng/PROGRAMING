#include <iostream>
using namespace std;

int main() {
    int N;
    cout << "Enter number of elements: ";
    cin >> N;

    int arr[N];
    cout << "Enter elements: ";
    for(int i = 0; i < N; i++) cin >> arr[i];

    // Selection sort
    for(int i = 0; i < N-1; i++) {
        int minIndex = i;
        for(int j = i+1; j < N; j++) {
            if(arr[j] < arr[minIndex]) minIndex = j;
        }
        swap(arr[i], arr[minIndex]);
    }

    cout << "Sorted array (ascending): ";
    for(int i = 0; i < N; i++) cout << arr[i] << " ";
    cout << endl;

    return 0;
}
