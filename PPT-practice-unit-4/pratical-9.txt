#include <iostream>
#include <fstream>
using namespace std;

bool isVowel(char ch) {
    ch = tolower(ch);
    return ch=='a'||ch=='e'||ch=='i'||ch=='o'||ch=='u';
}

int main() {
    ifstream inFile("example.txt");
    if(!inFile) { cout << "Cannot open file.\n"; return 1; }

    char ch;
    int upper=0, lower=0, digits=0, vowels=0, special=0;

    while(inFile.get(ch)) {
        if(isupper(ch)) upper++;
        else if(islower(ch)) lower++;
        if(isdigit(ch)) digits++;
        if(isVowel(ch)) vowels++;
        if(!isalnum(ch) && !isspace(ch)) special++;
    }

    inFile.close();
    cout << "Uppercase: " << upper << "\nLowercase: " << lower
         << "\nDigits: " << digits << "\nVowels: " << vowels
         << "\nSpecial symbols: " << special << endl;

    return 0;
}
