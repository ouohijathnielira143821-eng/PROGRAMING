#include <iostream>
#include <string>
using namespace std;

int main() {
    string str;
    cout << "Enter a string: ";
    cin >> str;

    string rev = str;
    reverse(rev.begin(), rev.end());

    if(str == rev) cout << "String is a palindrome.\n";
    else cout << "String is not a palindrome.\n";

    return 0;
}
