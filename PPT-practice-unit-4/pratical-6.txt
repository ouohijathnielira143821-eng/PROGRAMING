#include <iostream>
#include <fstream>
using namespace std;

int main() {
    ofstream data("DATA.txt");
    for(int i = 1; i <= 10; i++) data << i << " ";
    data.close();

    ifstream inFile("DATA.txt");
    ofstream odd("ODD.txt"), even("EVEN.txt");

    int num;
    while(inFile >> num) {
        if(num % 2 == 0) even << num << " ";
        else odd << num << " ";
    }

    inFile.close();
    odd.close();
    even.close();
    cout << "Numbers separated into ODD.txt and EVEN.txt\n";
    return 0;
}
