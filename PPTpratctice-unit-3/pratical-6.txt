#include <iostream>
using namespace std;

int main() {
    int N;
    cout << "Enter number of elements: ";
    cin >> N;

    int *arr = new int[N];
    int sumEven = 0, sumOdd = 0;

    cout << "Enter elements: ";
    for(int i = 0; i < N; i++) {
        cin >> arr[i];
        if(arr[i] % 2 == 0) sumEven += arr[i];
        else sumOdd += arr[i];
    }

    cout << "Sum of even numbers = " << sumEven << endl;
    cout << "Sum of odd numbers = " << sumOdd << endl;

    delete[] arr;
    return 0;
}
