#include <iostream>
using namespace std;

int main() {
    int N;
    cout << "Enter number of elements: ";
    cin >> N;

    int arr[N];
    cout << "Enter elements: ";
    for(int i = 0; i < N; i++) cin >> arr[i];

    // Shell sort
    for(int gap = N/2; gap > 0; gap /= 2) {
        for(int i = gap; i < N; i++) {
            int temp = arr[i];
            int j;
            for(j = i; j >= gap && arr[j-gap] > temp; j -= gap) {
                arr[j] = arr[j-gap];
            }
            arr[j] = temp;
        }
    }

    cout << "Sorted array (ascending): ";
    for(int i = 0; i < N; i++) cout << arr[i] << " ";
    cout << endl;

    return 0;
}
