#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    int N, key;
    cout << "Enter number of elements: ";
    cin >> N;

    int arr[N];
    cout << "Enter elements: ";
    for(int i = 0; i < N; i++) cin >> arr[i];

    sort(arr, arr + N);  // ensure array is sorted
    cout << "Enter value to search: ";
    cin >> key;

    int low = 0, high = N-1, pos = -1;
    while(low <= high) {
        int mid = (low + high) / 2;
        if(arr[mid] == key) {
            pos = mid;
            break;
        } else if(arr[mid] < key) {
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }

    if(pos != -1)
        cout << "Element found at position " << pos << endl;
    else
        cout << "Element not found.\n";

    return 0;
}
