#include <iostream>
using namespace std;

int main() {
    int r, c;
    cout << "Enter rows and columns: ";
    cin >> r >> c;

    int *A[r], *B[r], *C[r];
    for(int i = 0; i < r; i++) {
        A[i] = new int[c];
        B[i] = new int[c];
        C[i] = new int[c];
    }

    cout << "Enter matrix A:\n";
    for(int i = 0; i < r; i++)
        for(int j = 0; j < c; j++) cin >> A[i][j];

    cout << "Enter matrix B:\n";
    for(int i = 0; i < r; i++)
        for(int j = 0; j < c; j++) cin >> B[i][j];

    // Add matrices
    for(int i = 0; i < r; i++)
        for(int j = 0; j < c; j++)
            C[i][j] = A[i][j] + B[i][j];

    cout << "Sum of matrices:\n";
    for(int i = 0; i < r; i++) {
        for(int j = 0; j < c; j++) cout << C[i][j] << " ";
        cout << endl;
    }

    for(int i = 0; i < r; i++) {
        delete[] A[i];
        delete[] B[i];
        delete[] C[i];
    }

    return 0;
}
