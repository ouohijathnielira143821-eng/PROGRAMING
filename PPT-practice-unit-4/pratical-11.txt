#include <iostream>
#include <fstream>
using namespace std;

int main() {
    ifstream file("example.txt", ios::binary | ios::ate);
    if(!file) { cout << "Cannot open file.\n"; return 1; }

    streamsize size = file.tellg();
    cout << "Size of file: " << size << " bytes\n";
    file.close();
    return 0;
}
