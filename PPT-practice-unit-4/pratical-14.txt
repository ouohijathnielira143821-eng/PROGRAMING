#include <iostream>
#include <fstream>
using namespace std;

int main(int argc, char* argv[]) {
    if(argc < 3) {
        cout << "Usage: ./program source.txt dest.txt\n";
        return 1;
    }

    ifstream src(argv[1], ios::binary);
    ofstream dest(argv[2], ios::binary);

    if(!src || !dest) { cout << "Error opening files.\n"; return 1; }

    char ch;
    while(src.get(ch)) dest.put(ch);

    src.close();
    dest.close();

    cout << "File copied successfully.\n";
    return 0;
}
